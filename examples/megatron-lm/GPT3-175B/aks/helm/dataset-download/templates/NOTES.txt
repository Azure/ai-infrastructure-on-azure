1. Get the status of your dataset download job with:
   kubectl get job {{ include "megatron-dataset-download.fullname" . }}

2. Follow the logs with:
   kubectl logs -f job/{{ include "megatron-dataset-download.fullname" . }}

3. Once completed, verify the downloaded data:
   kubectl run verify-data --rm -i --tty --image={{ .Values.image.repository }}:{{ .Values.image.tag }} -- \
     bash -c "ls -la {{ .Values.storage.mount }}/{{ .Values.dataset.outputPath }}/ | head -20"

4. Check total dataset size:
   kubectl run check-size --rm -i --tty --image={{ .Values.image.repository }}:{{ .Values.image.tag }} -- \
     bash -c "du -sh {{ .Values.storage.mount }}/{{ .Values.dataset.outputPath }}"

{{- if not .Values.dataset.fullDataset }}
Note: You downloaded a sample dataset with {{ .Values.dataset.sampleFiles }} files.
To download the full dataset, set dataset.fullDataset=true
{{- end }}
